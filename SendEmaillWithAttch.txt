// Clean XML
Func<string, string> CleanXml = rawXml =>
{
    if (string.IsNullOrWhiteSpace(rawXml))
        throw new Exception("XML is empty");

    rawXml = rawXml.Trim('\uFEFF', '\u200B');

    int xmlStart = rawXml.IndexOf("<?xml", StringComparison.OrdinalIgnoreCase);
    if (xmlStart > 0)
        rawXml = rawXml.Substring(xmlStart);

    rawXml = rawXml.Replace("“", "\"").Replace("”", "\"");
    return rawXml.Trim();
};
